      def get_args():

          parser = optparse.OptionParser()
          parser.add_option("-i", "--interface", dest="interface", help="select interface")
          parser.add_option("-m", "--mac", dest="new_mac", help="Enter new MAC address")
          parser.add_option("--show all", dest="show_int", help="Display all network interfaces.")
          (opts, args) = parser.parse_args()


          if not opts.interface:
               parser.error("Please input interface, type --help for more details")
          elif not opts.new_mac:
              parser.error("Please input new MAC address, type --help for more details")

          return opts, args

      def change_mac(interface, new_mac):
          subprocess.call(["ifconfig", interface, "down"])
          subprocess.call(["ifconfig", interface, "hw", "ether", new_mac])
          subprocess.call(["ifconfig", interface, "up", ])
          print("[+] Changing MAC Address to " + new_mac)


      def get_current_mac(interface):
          show_ifconfig = subprocess.check_output(["ifconfig", interface])
          mac_regex = '\w\w:\w\w:\w\w:\w\w:\w\w:\w\w'
          ifconfig_mac = re.search(mac_regex, str(show_ifconfig))
          if ifconfig_mac:
              return ifconfig_mac.group(0)

          else:
              print("Could not read MAC Address.")


      current_mac = get_current_mac(opts.interface)
      print("Current MAC Address " + str(current_mac))
      change_mac(opts.interface, opts.new_mac)

      current_mac = get_current_mac(opts.interface)
      if current_mac == opts.new_mac:
